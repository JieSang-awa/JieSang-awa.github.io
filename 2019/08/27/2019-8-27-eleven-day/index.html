简单命令

JieSangQAQ 发布于 2019-08-27

echo //显示文本行

echo $LANG //显示系统语言

echo $PATH //显示系统的环境变量

date //显示当前时间

date +’%Y/%m/%d’ //以yyyy/mm/dd格式显示日期

date +’%Y-%m-%d’ //以yyyy-mm-dd格式显示日期

date +%H:%M //显示时、分

date -r test //显示test文件最后一次的修改时间

cat /etc/shells //查看shell版本

hostname //显示主机名

hostname -a //显示主机别名

hostname -d //显示主机域名

hostname -i //显示主机IP地址

hostname 主机名 //设置主机名称

cal //显示当前月份

查看Linux 端口

netstat -pan|grep 177 //查看177端口情况

netstat -nlap //查询进程占用哪些端口
